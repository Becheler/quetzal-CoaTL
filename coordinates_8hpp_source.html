156 constexpr double distanceEarth(double lat1d, double lon1d, double lat2d, double lon2d)
157 {
158 
159  const double earthRadiusKm = 6371.0;
160 
161  const double lat1r = deg2rad(lat1d);
162  const double lon1r = deg2rad(lon1d);
163  const double lat2r = deg2rad(lat2d);
164  const double lon2r = deg2rad(lon2d);
165 
166  const double u = sin((lat2r - lat1r) / 2);
167  const double v = sin((lon2r - lon1r) / 2);
168 
169  const double d = 2.0 * earthRadiusKm * asin(sqrt(u * u + cos(lat1r) * cos(lat2r) * v * v));
170  assert(d >= 0);
171  return d;
172 }
