78  class JC69
79  {
80  static const std::map<char, std::string> dico;
81  public:
82  // @typedef allelic state type
83  using state_type = char;
84  /*
85  * @brief Sample a new mutational state
86  * @param state the initial mutational state
87  * @param nb_mutations the number of mutational events to simulate
88  * @param gen a random number generator
89  * @tparam Generator a random number generator
90  * @return
91  */
92  template<typename Generator>
93  static state_type mute(state_type state, unsigned int nb_mutations, Generator & gen)
94  {
95  assert( dico.find(state) != dico.end() );
96  for(unsigned int i = 0; i < nb_mutations; ++i)
97  {
98  std::uniform_int_distribution dist(0,2);
99  state = dico.at(state)[dist(gen)];
100  }
101  return state;
102  }
