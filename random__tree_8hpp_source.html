54  template<class Graph=quetzal::coalescence::k_ary_tree<boost::no_property,boost::no_property>, class Generator>
55  auto get_random_spanning_tree(int n_vertices, int n_edges, Generator& rng)
56  {
57  using UG = detail::Undirect<Graph>;
58  using vertex_t = typename UG::vertex_descriptor;
59 
60  // assuming integral vertex index for simplicity
61  static_assert(std::is_same_v<vertex_t, size_t>);
62  static_assert(std::is_same_v<typename UG::vertex_descriptor,
63  typename Graph::vertex_descriptor>);
64 
65  // Initialize a random undirected graph
66  UG ug;
67  generate_random_graph(ug, n_vertices, n_edges, rng);
68  vertex_t const root = random_vertex(ug, rng);
69 
70  // make the graph fully connected
71  {
72  std::map<vertex_t, int> components;
73  auto from = [&](int component) { // just picking the first...
74  for (auto& [v, c] : components) if (c == component) return v;
75  throw std::range_error("component");
76  };
77 
78  // records the component number in the component property map
79  auto cmap = boost::make_assoc_property_map(components);
80  // The algorithm computes how many connected components are in the graph
81  if (int n = connected_components(ug, cmap); n > 1)
82  {
83  for (int c = 1; c < n; ++c)
84  add_edge(from(c - 1), from(c), ug);
85  }
86  }
87 
88  std::map<vertex_t, vertex_t> predecessors;
89  random_spanning_tree(ug, rng,
90  boost::root_vertex(root) //
91  .predecessor_map(boost::make_assoc_property_map(predecessors)));
92 
93  Graph tree(num_vertices(ug)); // build a tree copy
94  for (auto v : boost::make_iterator_range(vertices(ug)))
95  if (predecessors.contains(v))
96  if (auto pred = predecessors.at(v); ug.null_vertex() != pred)
97  add_edge(predecessors.at(v), v, tree);
98 
99  return std::tuple(std::move(tree), root);
100  }
