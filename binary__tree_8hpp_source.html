38template <typename T> struct tag
39{
40 using type = T;
41};

43template <typename F> struct initializer
44{
45 F f;
46 template <typename T> operator T() &&
47 {
48 return std::forward<F>(f)(tag<T>{});
49 }
50};

52template <typename F> initializer(F &&) -> initializer<F>;
53
54template <typename... Args> auto initpack(Args &&...args)
55{
56 return initializer{[&](auto t) {
57 using Ret = typename decltype(t)::type;
58 return Ret{std::forward<Args>(args)...};
59 }};
60}

62template <typename VertexDescriptor, typename VertexProperty> struct VertexManager
63{
64 using vertex_descriptor = VertexDescriptor;
65 using vertex_hashmap_type = std::map<vertex_descriptor, VertexProperty>;
66
67 vertex_hashmap_type _property_map;
68
69 template <typename... Args> void add_vertex_to_manager(vertex_descriptor v, Args &&...args)
70 {
71 _property_map[v] = {std::forward<Args>(args)...};
72 }
73
74 const VertexProperty &operator[](vertex_descriptor v) const
75 {
76 return _property_map.at(v);
77 }
78
79 VertexProperty &operator[](vertex_descriptor v)
80 {
81 return _property_map[v];
82 }

85template <typename EdgeDescriptor, typename EdgeProperty> struct EdgeManager
86{
87 using edge_descriptor = EdgeDescriptor;
88 using vertex_descriptor = typename EdgeDescriptor::first_type;
89 using edge_hashmap_type = std::map<edge_descriptor, EdgeProperty>;
90
91 edge_hashmap_type _property_map;
92
93 void add_edges(EdgeDescriptor u, const EdgeProperty &left, EdgeDescriptor v, const EdgeProperty &right)
94 {
95 _property_map.insert({u, left});
96 _property_map.insert({v, right});
97 }
98
99 template <typename... Args>
100 void add_edges(EdgeDescriptor u, std::tuple<Args...> left, EdgeDescriptor v, std::tuple<Args...> right)
101 {
102 _property_map.emplace(
103 std::piecewise_construct, std::forward_as_tuple(u),
104 std::forward_as_tuple(std::apply([](auto &&...args) { return initpack(args...); }, left)));
105 _property_map.emplace(
106 std::piecewise_construct, std::forward_as_tuple(v),
107 std::forward_as_tuple(std::apply([](auto &&...args) { return initpack(args...); }, right)));
108 }
109
110 template <typename VertexDescriptor>
111 const EdgeProperty &operator[](const std::pair<VertexDescriptor, VertexDescriptor> &edge) const
112 {
113 return _property_map.at(edge);
114 }
115
116 template <typename VertexDescriptor>
117 EdgeProperty &operator[](const std::pair<VertexDescriptor, VertexDescriptor> &edge)
118 {
119 return _property_map.at(edge);
120 }

689template <typename Vertex, typename Edge, typename Generator>
690auto update_tree(binary_tree<Vertex, Edge> &tree,
691 std::vector<typename binary_tree<Vertex, Edge>::vertex_descriptor> leaves, Generator &rng)
692{
693 using tree_type = binary_tree<Vertex, Edge>;
694 using vertex_descriptor = typename tree_type::vertex_descriptor;
695
696 if (leaves.size() == 1)
697 {
698 return leaves.front();
699 }
700 else
701 {
702 std::uniform_int_distribution<> distrib(1, leaves.size() - 1);
703 int split = distrib(rng);
704
705 std::vector<vertex_descriptor> left(leaves.begin(), leaves.begin() + split);
706 std::vector<vertex_descriptor> right(leaves.begin() + split, leaves.end());
707
708 vertex_descriptor parent;
709
710 if constexpr (std::is_same_v<Vertex, boost::no_property>)
711 parent = tree.add_vertex();
712 else
713 parent = tree.add_vertex(Vertex());
714
715 if constexpr (std::is_same_v<Edge, boost::no_property>)
716 tree.add_edges(parent, update_tree(tree, left, rng), update_tree(tree, right, rng));
717 else
718 tree.add_edges(parent, {update_tree(tree, left, rng), Edge()}, {update_tree(tree, right, rng), Edge()});
719 return parent;
720 }
721}

729template <typename Vertex = boost::no_property, typename Edge = boost::no_property, typename Generator>
730std::pair<quetzal::coalescence::binary_tree<Vertex, Edge>,
731 typename quetzal::coalescence::binary_tree<Vertex, Edge>::vertex_descriptor>
732get_random_binary_tree(int n_leaves, Generator &rng)
733{
734 using tree_type = quetzal::coalescence::binary_tree<Vertex, Edge>;
735 using vertex_descriptor = typename tree_type::vertex_descriptor;
736
737 tree_type tree;
738
739 std::vector<vertex_descriptor> leaves(n_leaves);
740
741 std::transform(leaves.cbegin(), leaves.cend(), leaves.begin(), [&tree](auto) {
742 if constexpr (std::is_same_v<Vertex, boost::no_property>)
743 return tree.add_vertex();
744 else
745 return tree.add_vertex(Vertex());
746 });
747
748 vertex_descriptor root = detail::update_tree(tree, leaves, rng);
749 return std::make_pair(std::move(tree), root);
750}
