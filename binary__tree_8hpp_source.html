108  template <typename... Args>
109  void
110  add_edges(EdgeDescriptor u, std::tuple<Args...> left, EdgeDescriptor v, std::tuple<Args...> right)
111  {
112  _property_map.emplace(std::piecewise_construct,
113  std::forward_as_tuple(u),
114  std::forward_as_tuple(std::apply([](auto &&...args)
115  { return initpack(args...); },
116  left)));
117  _property_map.emplace(std::piecewise_construct,
118  std::forward_as_tuple(v),
119  std::forward_as_tuple(std::apply([](auto &&...args)
120  { return initpack(args...); },
121  right)));
122  }

681  template <typename Vertex, typename Edge, typename Generator>
682  auto update_tree(
683  binary_tree<Vertex, Edge> &tree,
684  std::vector<typename binary_tree<Vertex, Edge>::vertex_descriptor> leaves,
685  Generator &rng)
686  {
687  using tree_type = binary_tree<Vertex, Edge>;
688  using vertex_descriptor = typename tree_type::vertex_descriptor;
689 
690  if (leaves.size() == 1)
691  {
692  return leaves.front();
693  }
694  else
695  {
696  std::uniform_int_distribution<> distrib(1, leaves.size() - 1);
697  int split = distrib(rng);
698 
699  std::vector<vertex_descriptor> left(leaves.begin(), leaves.begin() + split);
700  std::vector<vertex_descriptor> right(leaves.begin() + split, leaves.end());
701 
702  vertex_descriptor parent;
703 
704  if constexpr (std::is_same_v<Vertex, boost::no_property>)
705  parent = tree.add_vertex();
706  else
707  parent = tree.add_vertex(Vertex());
708 
709  if constexpr (std::is_same_v<Edge, boost::no_property>)
710  tree.add_edges(parent, update_tree(tree, left, rng), update_tree(tree, right, rng));
711  else
712  tree.add_edges(parent,
713  { update_tree(tree, left, rng), Edge() },
714  { update_tree(tree, right, rng), Edge()});
715  return parent;
716  }
717  }
718  }

726  std::pair< quetzal::coalescence::binary_tree<Vertex, Edge>, typename quetzal::coalescence::binary_tree<Vertex, Edge>::vertex_descriptor>
727  get_random_binary_tree(int n_leaves, Generator &rng)
728  {
729  using tree_type = quetzal::coalescence::binary_tree<Vertex, Edge>;
730  using vertex_descriptor = typename tree_type::vertex_descriptor;

733 
734  std::vector<vertex_descriptor> leaves(n_leaves);

740  return tree.add_vertex();

742  return tree.add_vertex(Vertex());
