286  template <typename Generator>
287  static auto update_tree(
288  CRTP &tree,
289  std::vector<vertex_descriptor> leaves,
290  Generator &rng)
291  {
292  using tree_type = CRTP;
293 
294  if (leaves.size() == 1)
295  {
296  return leaves.front();
297  }
298  else
299  {
300  std::uniform_int_distribution<> distrib(1, leaves.size() - 1);
301  int split = distrib(rng);
302 
303  std::vector<vertex_descriptor> left(leaves.begin(), leaves.begin() + split);
304  std::vector<vertex_descriptor> right(leaves.begin() + split, leaves.end());
305 
306  auto parent = boost::add_vertex(tree._graph);
307  if constexpr (std::is_same_v<EdgeProperty, boost::no_property>)
308  tree.add_edges(parent, { update_tree(tree, left, rng), update_tree(tree, right, rng) });
309  else
310  tree.add_edges(parent,
311  { std::make_tuple(update_tree(tree, left, rng), EdgeProperty()),
312  std::make_tuple(update_tree(tree, right, rng), EdgeProperty()) } );
313  return parent;
314  }

677  template <typename Vertex, typename Edge, typename Generator>
678  auto update_tree(
679  k_ary_tree<Vertex, Edge> &tree,
680  std::vector<typename k_ary_tree<Vertex, Edge>::vertex_descriptor> leaves,
681  Generator &rng)
682  {
683  using tree_type = k_ary_tree<Vertex, Edge>;
684  using vertex_descriptor = typename tree_type::vertex_descriptor;
685 
686  if (leaves.size() == 1)
687  {
688  return leaves.front();
689  }
690  else
691  {
692  std::uniform_int_distribution<> distrib(1, leaves.size() - 1);
693  int split = distrib(rng);
694 
695  std::vector<vertex_descriptor> left(leaves.begin(), leaves.begin() + split);
696  std::vector<vertex_descriptor> right(leaves.begin() + split, leaves.end());
697 
698  vertex_descriptor parent;
699 
700  if constexpr (std::is_same_v<Vertex, boost::no_property>)
701  parent = tree.add_vertex();
702  else
703  parent = tree.add_vertex(Vertex());
704 
705  if constexpr (std::is_same_v<Edge, boost::no_property>)
706  tree.add_edges(parent, { update_tree(tree, left, rng), update_tree(tree, right, rng) } );
707  else
708  tree.add_edges(parent, { { update_tree(tree, left, rng), Edge() }, { update_tree(tree, right, rng), Edge()} } );
709  return parent;
710  }
711  }
