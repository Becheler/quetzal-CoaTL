39  bool check_if_balanced(const std::string & input, const char & open='(', const char & close=')')
40  {
41  int count = 0;
42  for(const auto & ch : input)
43  {
44  if (ch == open ) count++;
45  if (ch == close ) count--;
46  // if a parenthesis is closed without being opened return false
47  if(count < 0)
48  return false;
49  }
50  // in the end the test is passed only if count is zero
51  return count == 0;
52  }

135  static std::string edit(const std::string& s)
136  {
137  std::string buffer;
138  int counter = 0;
139  for (const auto & ch : s)
140  {
141  if (ch == '[' ) counter++;
142  if (ch == ']' ) counter--;
143  if ( ch == '[' && counter == 2) continue; // do nothing, that was the opening
144  if ( ch == ']' && counter == 1) continue; // do nothing, that was the closing
145  if ( !( counter >=2 || (counter == 1 && ch == ']') ) ) buffer.append(std::string(1, ch));
146  }
147  return buffer;
148  }
