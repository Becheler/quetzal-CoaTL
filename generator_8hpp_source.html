57  bool check_if_balanced(std::string_view input, const char & open='(', const char & close=')')
58  {
59  int count = 0;
60  for(const auto & ch : input)
61  {
62  if (ch == open ) count++;
63  if (ch == close ) count--;
64  // if a parenthesis is closed without being opened return false
65  if(count < 0)
66  return false;
67  }
68  // in the end the test is passed only if count is zero
69  return count == 0;
70  }

162  template<> struct remove_comments_of_depth<2>
163  {
164  static std::string edit(const std::string s)
165  {
166  std::string buffer;
167  int counter = 0;
168  for (const auto & ch : s)
169  {
170  if (ch == '[' ) counter++;
171  if (ch == ']' ) counter--;
172  if ( ch == '[' && counter == 2) continue; // do nothing, that was the opening
173  if ( ch == ']' && counter == 1) continue; // do nothing, that was the closing
174  if ( !( counter >=2 || (counter == 1 && ch == ']') ) ) buffer.append(std::string(1, ch));
175  }
176  return buffer;
177  }
178  };
