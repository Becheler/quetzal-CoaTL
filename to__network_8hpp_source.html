22  template<class Graph>
23  void regraft_cycles(Graph& graph, auto root)
24  {
25  using graph_type = Graph;
26  using vertex_descriptor = typename graph_type::vertex_descriptor;
27  using vertex_property = typename Graph::vertex_property;
28  using dict_type = std::map<std::string, std::map<std::string, std::vector<vertex_descriptor>>>;
29 
30  dict_type records;
31  std::regex rgx(
32  "[a-zA-Z0-9_]*" // Label: only upper and lowercase letters, numbers, underscores, or empty
33  "#" // Cycle detected
34  "([H|LGT|R])" // Hybrid, Lateral Gene Transfer or Recombination
35  "(0|[1-9][0-9]*)" // Any positive integer, including 0
36  );
37 
38  // iterate through vertices/labels
39  for (auto vertex : graph.vertices() )
40  {
41  std::smatch match;
42  std::string label;
43  if constexpr (std::is_same_v<vertex_property, std::string>)
44  label = graph[vertex];
45  else
46  label = graph[vertex].label();
47 
48  // find if cycle detected
49  if(std::regex_search(label, match, rgx))
50  {
51  // save vertices id to their rightful groups
52  const auto event_type = match.str(1);
53  const auto event_id = match.str(2);
54  records[event_type][event_id].push_back(vertex);
55  }
56  }
57 
58  using namespace ranges;
59 
60  // iterate through results
61  for( auto const& [event_type, val] : records )
62  {
63  for(auto const& [event_id, duplicated_vertices] : val)
64  {
65  auto new_vertex = graph.add_vertex("#" + event_type + event_id);
66 
67  auto new_sources = duplicated_vertices
68  | views::transform([&graph](auto v){return graph.in_edges(v);})
69  | views::take(1)
70  | views::transform([&graph](auto&& range_of_e){return graph.source(*range_of_e.begin());});
71 
72  auto new_targets = duplicated_vertices
73  | views::transform([&graph](auto v){return graph.out_edges(v);})
74  | views::join
75  | views::transform([&graph](auto e){return graph.target(e);});
76 
77  ranges::for_each(duplicated_vertices, [&graph](auto v){graph.clear_vertex(v); graph.remove_vertex(v);});
78 
79  auto regraft = [&graph, new_vertex](auto s, auto t){
80  graph.add_edge(s, new_vertex);
81  graph.add_edge(new_vertex, t);
82  };
83 
84  auto todo = ranges::views::zip(new_sources, new_targets);
85  ranges::for_each(todo, [regraft](auto&& z){ std::apply(regraft, z);});
86  }
87  }
88  }
89  } // end namespace detail

92  template
93  <
94  class VertexProperty = quetzal::format::newick::ast::node::vertex_property,
95  class EdgeProperty= quetzal::format::newick::ast::node::edge_property
96  >
97  std::tuple
98  <
99  quetzal::coalescence::network<VertexProperty,EdgeProperty>,
100  typename quetzal::coalescence::network<VertexProperty,EdgeProperty>::vertex_descriptor
101  >
102  to_network(quetzal::format::newick::ast::node ast_root)
103  {
104  using graph_type = quetzal::coalescence::network<VertexProperty,EdgeProperty>;
105  using vertex_descriptor = typename graph_type::vertex_descriptor;
106 
107  graph_type graph;
108  auto root = graph.add_vertex( VertexProperty(ast_root.name) );
109 
110  // A hacky recursive lambda (C++23 will make this much simpler)
111  static const auto populate = [](graph_type& graph, vertex_descriptor parent, const auto& ast)
112  {
113  static auto recursive = [](graph_type& graph, vertex_descriptor parent, const auto& ast, auto& self) mutable -> void
114  {
115  std::vector<std::tuple< vertex_descriptor, EdgeProperty>> children;
116 
117  children.reserve(ast.children.size());
118 
119  for(const auto& ast_child : ast.children){
120  children.push_back(std::make_tuple( graph.add_vertex( VertexProperty(ast_child.name) ), EdgeProperty(ast_child.distance_to_parent) ));
121  }
122 
123  graph.add_edges(parent, children);
124 
125  for ( int i = 0; i < children.size(); i++){
126  self(graph, std::get<0>(children[i]), ast.children[i], self);
127  }
128 
129  }; // end recursive
130  recursive(graph, parent, ast, recursive);
131  }; // end populate
132 
133  populate(graph, root, ast_root);
134  detail::regraft_cycles(graph, root);
135  return {std::move(graph), root};
136  }
