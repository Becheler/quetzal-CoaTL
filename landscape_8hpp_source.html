74  landscape(const std::map<key_type, std::filesystem::path> &files, const std::vector<time_type> & times) :
75  _variables()
76  {
77 
78  for (auto const &it : files)
79  {
80  _variables.emplace(std::piecewise_construct, std::forward_as_tuple(it.first), std::forward_as_tuple(it.second, times));
81  }
82 
83  auto have_same_origin = [this](auto const &it)
84  { return it.second.origin() == _variables.cbegin()->second.origin(); };
85 
86  if (!all_of(_variables.cbegin(), _variables.cend(), have_same_origin))
87  {
88  throw std::runtime_error("all ecological quantities dataset must have same origin");
89  }
90 
91  auto have_same_extent = [this](auto const &it)
