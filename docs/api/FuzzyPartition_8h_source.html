119  FuzzyPartition & merge_clusters(quetzal::fuzzy_transfer_distance::RestrictedGrowthString const& rgs)
120  {
121  assert( rgs.size() == this->nClusters() );
122  matrix_type new_coefficients(rgs.nBlocks(), this->nElements(), 0);
124  for(unsigned int i = 0; i < m_membership_coefficients.size1(); ++i){
125  for(unsigned int j = 0; j < m_membership_coefficients.size2(); ++j){
126  new_coefficients(rgs.at(i), j) += m_membership_coefficients(i,j);
127  }
128  }
129  m_membership_coefficients.swap(new_coefficients);
130  std::set<cluster_type> c;
131  for(unsigned int i = 0; i < rgs.nBlocks(); ++i){
132  c.insert(i);

175  template<typename V, typename W>
176  double compute_cost(V const& v, W const& w) const {
177  assert(v.size() == w.size());
178 
179  std::vector<double> distances;
180 
181  for(auto && t : zip_range(v, w) ) {
182  distances.push_back(std::abs( boost::get<0>(t) - boost::get<1>(t) ));
183  }
184 
185  return std::accumulate(distances.begin(), distances.end(), 0.0);
186  }

197  template<typename T>
198  double compute_solution_cost(Matrix<T> const& solution, Matrix<T> const& costs) const {
199  double total_cost = 0.0;
200  for (unsigned int i = 0 ; i < solution.rows(); ++i ) {
201  int row_matches = 0;
202  for ( unsigned int j = 0 ; j < solution.columns() ; ++j ) {
203  if ( solution(i,j) == 0 ){
204  ++row_matches;
205  total_cost += costs(i,j);
206  }
207  }
208  assert(row_matches == 0 || row_matches == 1);
209  }
210  return total_cost;
211  }

231
