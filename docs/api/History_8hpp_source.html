68  void simulate_forward(T sim_growth, U kernel, V& gen)
69  {
70  for(unsigned int t = 0; t < this->nb_generations(); ++t)
71  {
72  unsigned int landscape_individuals_count = 0;
73  for(auto x : this->distribution_area(t) )
74  {
75  auto N_tilde = sim_growth(gen, x, t);
76  landscape_individuals_count += N_tilde;
77  if(N_tilde >= 1)
78  {
79  for(unsigned int ind = 1; ind <= N_tilde; ++ind)
80  {
81  auto y = kernel(gen, x);
82  this->m_flows->add_to_flow_from_to(x, y, t, 1);
83  this->pop_size(y, t+1) += 1;
84  }
85  }
86  }
87  if(landscape_individuals_count == 0)
88  {
89  throw std::domain_error("Landscape populations went extinct before sampling");
90  }
91  }
92  }

128  {

130  {
131  unsigned int landscape_individuals_count = 0;
132  for(auto x : this->distribution_area(t) )
133  {
134  auto N_tilde = sim_growth(gen, x, t);
135  for(auto const& y : kernel.arrival_space(x) )
136  {
137  auto m = kernel(x, y);
138  assert(m >= 0.0 && m <= 1.0);
139  // Use of std::ceil to allow oversea migration
140  double nb_migrants = std::ceil(m * static_cast<double>(N_tilde));
141  if(nb_migrants >= 0){
142  landscape_individuals_count += nb_migrants;
143  this->m_flows->set_flow_from_to(x, y, t, nb_migrants);
144  this->pop_size(y, t+1) += nb_migrants;
145  }
146  }
147  }
