59  struct on_the_fly
60  {
70  template<typename Generator>
71  static auto sample(int k, int N, Generator& g)
72  {
73  assert(k > 0);
74  assert(N > 0);
75  std::vector<int> M_j(k+1);
76  std::vector<int> parents(N);
77  // Populate the urns (parents) by throwing balls (lineages) at them
78  std::uniform_int_distribution<int> distribution(0, N-1);
79  for(int i = 1; i <=k; ++i)
80  {
81  parents[distribution(g)] +=1;
82  }
83  // Build the occupancy spectrum by computing the occupancy numbers
84  for(int it : parents)
85  {
86  M_j[it] += 1;
87  }
88  assert(M_j.size() == k+1);
89  return OccupancySpectrum(M_j, k, N);
90  }
