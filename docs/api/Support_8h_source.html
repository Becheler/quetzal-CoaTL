72  template<class Callback>
73  void f(int n, int m, int j_max, std::vector<int>& v, Callback callback) const
74  {
75  if(m == 0 && n == 0)
76  {
77  // problem solved
78  callback(OccupancySpectrum(std::move(v), this->initial_n, this->initial_m));
79  return;
80  }
81  if (m != 0)
82  {
83  if (n == 0) {
84  auto copy = v;
85  copy[0] = m; // here n = 0 et m = 0, so problem is solved
86  callback(OccupancySpectrum(std::move(copy), this->initial_n, this->initial_m));
87  return;
88  } else {
89  if( j_max > 0){
90  // vertical descent
91  for(int i = n/j_max; i >= 1; --i)
92  {
93  auto copy = v;
94  copy[j_max] = i;
95  if(m >= i){
96  int balls_left = n - i*j_max;
97  if(balls_left < j_max){
98  f( balls_left, m-i, balls_left, copy, callback);
99  }else {
100  f( balls_left, m-i, j_max-1, copy, callback);
101  }
102  }
103  }
104  }
105  }
106  }
107  // horizontal descent
108  if( j_max == 0 && n > 0 )
109  {
110  // not solvable
111  return;
112  }
113  auto copy = v;
114  copy[j_max] = 0;
115  f(n, m, j_max-1, copy, callback);
116  }
