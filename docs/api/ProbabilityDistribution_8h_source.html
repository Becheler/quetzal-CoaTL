177  static double compute_probability(int n, int m, spectrum_type const& m_js)
178  {
179  BigFloat numerator(Factorial(m)*Factorial(n));
180  BigFloat denominator(1.0);
181  int j = 0;
182  for(auto m_j : m_js)
183  {
184  denominator *= boost::multiprecision::pow(BigFloat(Factorial(j)),BigFloat(m_j)) * BigFloat(Factorial(m_j));
185  ++j;
186  }
187  denominator *= BigFloat(std::pow(m,n));
188  BigFloat result = numerator/denominator;
189  return static_cast<double>(result) ;
