474  template <typename BinaryTree, typename Visitor>
475  Visitor traverse_nonempty(vertex_descriptor_t<BinaryTree> u, BinaryTree const &g, Visitor vis)
476  {
477  vis(visit::pre, u);
478  if (has_left_successor(u, g))
479  vis = traverse_nonempty(left_successor(u, g), g, vis);
480  vis(visit::in, u);
481  if (has_right_successor(u, g))
482  vis = traverse_nonempty(right_successor(u, g), g, vis);
483  vis(visit::post, u);
484  return vis;
485  }

488  template <typename BinaryTree>
489  int traverse_step(visit &stage, vertex_descriptor_t<BinaryTree> &u, BinaryTree const &g)
490  {
491  BOOST_CONCEPT_ASSERT((concepts::BidirectionalBinaryTreeConcept<BinaryTree>));
492 
493  switch (stage)
494  {
495 
496  case visit::pre:
497 
498  if (has_left_successor(u, g))
499  {
500  u = left_successor(u, g);
501  return 1;
502  }
503 
504  stage = visit::in;
505  return 0;
506 
507  case visit::in:
508 
509  if (has_right_successor(u, g))
510  {
511  stage = visit::pre;
512  u = right_successor(u, g);
513  return 1;
514  }
515 
516  stage = visit::post;
517  return 0;
518 
519  case visit::post:
520 
521  if (is_left_successor(u, g))
522  {
523  stage = visit::in;
524  }
525  u = predecessor(u, g);
526  return -1;
527 
528  default:
529  throw std::logic_error("Unexpected visit stage encountered");
530  }
531 
532  }

534  template <typename BinaryTree, typename Visitor>
535  Visitor traverse(vertex_descriptor_t<BinaryTree> u, BinaryTree const &g, Visitor vis)
536  {
537  if (empty(u, g))
538  return vis;
539 
540  auto root = u;
541 
542  visit stage = visit::pre;
543  vis(stage, u);
544 
545  do
546  {
547  traverse_step(stage, u, g);
548  vis(stage, u);
549  } while (u != root || stage != visit::post);
550 
551  return vis;
552  }

555  template <typename BinaryTree0, typename BinaryTree1>
556  bool bifurcate_isomorphic_nonempty(
557  vertex_descriptor_t<BinaryTree0> u, BinaryTree0 const &g,
558  vertex_descriptor_t<BinaryTree1> v, BinaryTree1 const &h)
559  {
560  BOOST_CONCEPT_ASSERT((concepts::ForwardBinaryTreeConcept<BinaryTree0>));
561  BOOST_CONCEPT_ASSERT((concepts::ForwardBinaryTreeConcept<BinaryTree1>));
562  BOOST_ASSERT(!empty(u, g));
563  BOOST_ASSERT(!empty(v, h));
564 
565  if (has_left_successor(u, g)) {
566  if (has_left_successor(v, h)) {
567  if (!bifurcate_isomorphic_nonempty(left_successor(u, g), g,
568  left_successor(v, h), h))
569  return false;
570  } else
571  return false;
572  } else if (has_left_successor(u, g))
573  return false;
574 
575  if (has_right_successor(u, g)) {
576  if (has_right_successor(v, h)) {
577  if (!bifurcate_isomorphic_nonempty(right_successor(u, g), g,
578  right_successor(v, h), h))
579  return false;
580  } else
581  return false;
582  } else if (has_right_successor(u, g))
583  return false;
584 
585  return true;
586  }

589  template <typename BinaryTree0, typename BinaryTree1>
590  bool bifurcate_isomorphic(
591  vertex_descriptor_t<BinaryTree0> u, BinaryTree0 const &g,
592  vertex_descriptor_t<BinaryTree1> v, BinaryTree1 const &h)
593  {
594  BOOST_CONCEPT_ASSERT((concepts::BidirectionalBinaryTreeConcept<BinaryTree0>));
595  BOOST_CONCEPT_ASSERT((concepts::BidirectionalBinaryTreeConcept<BinaryTree1>));
596 
597  if (empty(u, g)) return empty(v, h);
598  if (empty(v, h)) return false;
599  auto root0 = u;
600  visit visit0 = visit::pre;
601  visit visit1 = visit::pre;
602  while (true) {
603  traverse_step(visit0, u, g);
604  traverse_step(visit1, v, h);
605  if (visit0 != visit1) return false;
606  if (u == root0 && visit0 == visit::post) return true;
607  }
608  }
