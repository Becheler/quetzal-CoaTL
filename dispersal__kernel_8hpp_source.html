333 template <QuantityOf<isq::length> Distance = mp_units::quantity<mp_units::si::metre>> struct gaussian_mixture
334 {
336  struct param_type
337  {
339  Distance a1;
341  Distance a2;
343  double p;
344 
345  public:
346  };
347 
353  static constexpr auto pdf(Distance r, param_type const &param)
354  {
355  using std::exp;
356  Distance a1 = param.a1;
357  Distance a2 = param.a2;
358  double p = param.p;
359  assert(a1 > 0. * m && a2 > 0. * m && p > 0. && p < 1. && r >= 0. * m);
360  return p / (pi * a1 * a1) * exp(-(r * r) / (a1 * a1)) +
361  (1. - p) / (pi * a2 * a2) * exp(-(r * r) / (a2 * a2));
362  }
363 
367  static constexpr Distance mean_dispersal_distance(param_type const &param)
368  {
369  using std::pow;
370  Distance a1 = param.a1;
371  Distance a2 = param.a2;
372  double p = param.p;
373  assert(a1 > 0. * m && a2 > 0. * m && p > 0. && p < 1.);
374  return pow(pi, 0.5) * (p * a1 + (1. - p) * a2) / 2.;
375  }
