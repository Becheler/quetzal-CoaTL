213  using std::pow;
214  Distance a = p.a;
215  double b = p.b;
216  assert(a > 0. * m && b > 1. && r >= 0. * m);

225  {
226  using std::pow;
227  using std::tgamma;
228  Distance a = p.a;

232  {

234  }

328 template <QuantityOf<isq::length> Distance = mp_units::quantity<mp_units::si::metre>> struct gaussian_mixture
329 {
331  struct param_type
332  {
334  Distance a1;
336  Distance a2;
338  double p;
339 
340  public:
341  };
342 
348  static constexpr auto pdf(Distance r, param_type const &param)
349  {
350  using std::exp;
351  Distance a1 = param.a1;
352  Distance a2 = param.a2;
353  double p = param.p;
354  assert(a1 > 0. * m && a2 > 0. * m && p > 0. && p < 1. && r >= 0. * m);
355  return p / (M_PI * a1 * a1) * exp(-(r * r) / (a1 * a1)) +
356  (1. - p) / (M_PI * a2 * a2) * exp(-(r * r) / (a2 * a2));
357  }
358 
362  static constexpr Distance mean_dispersal_distance(param_type const &param)
363  {
364  using std::pow;
365  Distance a1 = param.a1;
366  Distance a2 = param.a2;
367  double p = param.p;
368  assert(a1 > 0. * m && a2 > 0. * m && p > 0. && p < 1.);
369  return pow(M_PI, 0.5) * (p * a1 + (1. - p) * a2) / 2.;
370  }
