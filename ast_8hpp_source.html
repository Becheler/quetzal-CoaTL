64  static inline std::ostream& print_subtree(node const& node, const std::string& prefix, std::ostream& os)
65  {
66  // if tree is more than a simple root node
67  if ( !node.children.empty())
68  {
69  // print a prefix growing with depth
70  os << prefix;
71 
72  size_t nb_children = node.children.size();
73 
74  // Bifurcation if more than 1 child
75  if (nb_children > 1) os << "├──";
76 
77  for (size_t i = 0; i < nb_children; ++i)
78  {
79  const auto& child = node.children[i];
80  // If not the last child
81  if (i < nb_children - 1)
82  {
83  // If not the first child
84  if (i > 0) os << prefix << "├──";
85 
86  bool should_print_branch = nb_children > 1 && !child.children.empty();
87  auto new_prefix = prefix + (should_print_branch ? "│\t" : "\t");
88  os << child.distance_to_parent << "──" << std::quoted(child.name) << "\n";
89  print_subtree(child, new_prefix, os);
90  }
91  // If the last child
92  else
93  {
94  if(nb_children > 1) os << prefix;
95  os << "└──" << child.distance_to_parent << "──" << std::quoted(child.name) << "\n";
96  print_subtree(child, prefix + "\t", os);
97  }
98  }
99  }
100  return os;
101  }
